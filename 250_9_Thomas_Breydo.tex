\begin{problem*}
Suppose $S,T\in\SL(V)$ and $ST$ is nilpotent. Prove that $TS$ is
nilpotent.
\end{problem*}

\vspace{0.5in}

\begin{claim*}
$TS$ is nilpotent.
\end{claim*}
\begin{proof}
Since $ST$ is nilpotent, then there exists some $k$ for which
\begin{align*}
    (ST)^{k} = 0.
\end{align*}
Note that
\begin{align*}
    (TS)^{k+1} &= T(ST)^kS \\
               &= 0,
\end{align*}
and thus $TS$ is nilpotent.
\end{proof}
